function pow(x,n)
{ // фигурную скобку перенести на первую строку
  let result=1;
  for(let i=0;i<n;i++) {result*=x;}
  return result;
}

let x=prompt("x?",''), n=prompt("n?",'')
if (n<=0)
{ // фигурную скобку поместить в конец предыдущей строки
  alert(`Степень ${n} не поддерживается, введите целую степень, большую 0`);
}
else // else поместить между закрывающей скобкой блока if и открывающей скобкой своего блока
{
  alert(pow(x,n))
}
